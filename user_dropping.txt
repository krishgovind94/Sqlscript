
Script to drop user from all DB’s at once.

EXEC sp_MSForEachDB 'USE [?];
        IF  EXISTS (SELECT * FROM sys.database_principals WHERE name = N''AMERUS\SV-CAUIM-DB'')
        DROP USER [AMERUS\SV-CAUIM-DB]; '

Delete login from server:

USE [master]
GO
DROP LOGIN [AMERUS\SV-CAUIM-DB]
GO
